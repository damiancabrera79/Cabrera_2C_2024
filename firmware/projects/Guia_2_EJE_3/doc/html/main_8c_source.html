289void app_main(void)
290{
291 LedsInit();
292 HcSr04Init(GPIO_3, GPIO_2); // GPIO_3: Echo | GPIO_2: Trigger
293 SwitchesInit();
294 LcdItsE0803Init();
295
296 SwitchActivInt(SWITCH_1, SwitchActivar, &activar);
297 SwitchActivInt(SWITCH_2, SwitchHold, &hold);
298
299 timer_config_t timer_A = {
300 .timer = TIMER_A,
301 .period = CONFIG_BLINK_PERIOD_uS,
302 .func_p = FuncionTimer,
303 .param_p = NULL};
304
305 TimerInit(&timer_A);
306
308 serial_config_t Uart = {
309 .port = UART_PC,
310 .baud_rate = 9600,
311 .func_p = FuncionUART,
312 .param_p = NULL
313 };
314
316 UartInit(&Uart);
317
318 xTaskCreate(&SensarTask, "Sensar", 512, NULL, 5, &sensar_task_handle);
319 xTaskCreate(&LedsTask, "Leds", 512, NULL, 5, &leds_task_handle);
320 xTaskCreate(&DisplayTask, "Display", 512, NULL, 5, &display_task_handle);
321 xTaskCreate(&EnviarDatosUART, "UART", 512, NULL, 4, &enviarDatosUART_task_handle);
322
323 // Iniciar el timer, con la estructura TIMER_A
324 TimerStart(timer_A.timer);
325}
