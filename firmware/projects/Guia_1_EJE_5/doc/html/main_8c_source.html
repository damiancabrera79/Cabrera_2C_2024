76void configurarGpios(uint8_t bcd_digito, gpioConf_t gpio_conf[4]) {
77 // Asegurarse de que el vector tiene al menos 4 elementos y que no es nulo
78 if (gpio_conf == NULL) {
79 return;
80 }
81
82 // Iterar sobre los 4 bits del dígito BCD (de b0 a b3)
83 for (uint8_t i = 0; i < 4; i++) {
84 // Configurar la dirección del GPIO (0 -> IN, 1 -> OUT)
85 GPIOInit(gpio_conf[i].pin, gpio_conf[i].dir);
86
87 // Verificar el estado del bit correspondiente
88 if (bcd_digito & (1 << i)) {
89 // Si el bit está en 1, configurar el GPIO en alto
90 GPIOOn(gpio_conf[i].pin);
91 } else {
92 // Si el bit está en 0, configurar el GPIO en bajo
93 GPIOOff(gpio_conf[i].pin);
94 }
95 }
96}
